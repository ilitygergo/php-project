CREATE TABLE users (
	id int NOT NULL AUTO_INCREMENT,
    first_name varchar(255),
    last_name varchar(255),
    email varchar(255),
    address varchar(255),
    gender varchar(255),
    age int,
    PRIMARY KEY (id)
);

CREATE TABLE products (
	id int NOT NULL AUTO_INCREMENT,
    brand varchar(255),
    cost int,
    category varchar(255),
    subcategory varchar(255),
	image varchar(255),
    PRIMARY KEY (id)
);
